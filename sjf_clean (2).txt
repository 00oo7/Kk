
#include <iostream>
using namespace std;

int findNextProcess(int at[], int bt[], bool done[], int k, int currentTime)
{
    int index = -1;
    int minBT = 1e9;
    for (int i = 0; i < k; i++)
    {
        if (!done[i] && at[i] <= currentTime)
        {
            if (bt[i] < minBT)
            {
                minBT = bt[i];
                index = i;
            }
        }
    }
    return index;
}

void calculateCT(int ct[], int at[], int bt[], int k)
{
    bool done[100] = {false};
    int completed = 0;
    int currentTime = 0;

    while (completed < k)
    {
        int idx = findNextProcess(at, bt, done, k, currentTime);

        if (idx == -1)
        {
            currentTime++;
        }
        else
        {
            currentTime += bt[idx];
            ct[idx] = currentTime;
            done[idx] = true;
            completed++;
        }
    }
}

int main()
{
    const int n = 10;
    int k;
    int bt[n], at[n], wt[n], tt[n], ct[n];

    cout << "Enter number of processes: ";
    cin >> k;

    for (int i = 0; i < k; i++)
    {
        cout << "Enter Arrival Time for Process " << i + 1 << ": ";
        cin >> at[i];
        cout << "Enter Burst Time for Process " << i + 1 << ": ";
        cin >> bt[i];
    }

    calculateCT(ct, at, bt, k);

    for (int i = 0; i < k; i++)
    {
        tt[i] = ct[i] - at[i];
        wt[i] = tt[i] - bt[i];
    }

    cout << "\nP#\tA.T\tB.T\tC.T\tT.T\tW.T\n";
    cout << "---------------------------------------------\n";

    for (int i = 0; i < k; i++)
    {
        cout << "P" << i + 1 << "\t" << at[i] << "\t" << bt[i]
             << "\t" << ct[i] << "\t" << tt[i] << "\t" << wt[i] << "\n";
    }

    float totalWT = 0, totalTT = 0;

    for (int i = 0; i < k; i++)
    {
        totalWT += wt[i];
        totalTT += tt[i];
    }

    cout << "\nAverage Waiting Time = " << totalWT / k << "\n";
    cout << "Average Turnaround Time = " << totalTT / k << "\n";

    return 0;
}
